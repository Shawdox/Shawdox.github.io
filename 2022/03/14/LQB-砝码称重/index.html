LQB-砝码称重

【LQB复习】砝码称重

​ 经典DP问题，维护一个可称量数组f[]，每次新加一个重量wi后将其与其中每个重量都相加减，同时保证结果都大于0即可：

#include "stdio.h"
#include "vector"
#include "algorithm"
using namespace std;
bool f[100001] = { 0, };
int main(void){
    int N,t;
    vector<int> now;
    scanf("%d", &N);
    for (int i = 0; i < N; ++i) {
        scanf("%d", &t);
        int size = now.size();//尤其注意这里，不能直接用now.size()，循环中push_back会改变其值。
        for (int j = 0; j < size; ++j) {//新加入的砝码与已称出的重量交互
            int a = now[j] + t, b = max(now[j], t) - min(t,now[j]);
            if (f[a] == false) {
                now.push_back(a);
                f[a] = true;
            }
            if (b != 0 && f[b] == false) {//注意称重量没有0
                now.push_back(b);
                f[b] = true;
            }
        }
        if (f[t] == false){//是否加入自己
            f[t] = true;
            now.push_back(t);
        }
    }
    printf("%d\n", now.size());
    return 0;
}
