算法-非零段划分

【算法复习】非零段划分

​ 第二遍重做，暴力的解法是O（n^2^），必不可能过，用浮水法来解，但只有70分，后来发现是ans的判断放到了双循环的最里面，小错误调了1h：

#include "stdio.h"
#include "vector"
#include "algorithm"
using namespace std;
int main(void){
    int n, A[5005] = { 0, },m=0,now = 0,ans = 0;
    vector<int> B[1005];//下标值索引
    scanf("%d", &n);
    for (int i = 1; i <= n; ++i) {
        scanf("%d", &A[i]);
        B[A[i]].push_back(i);
        m = max(m, A[i]);
    }
    for (int p = m; p >= 1; --p) {
        for (int i = 0; i < B[p].size(); ++i) {
            int left = B[p][i] - 1,right;
            while (i + 1 < B[p].size() && B[p][i] + 1 == B[p][i + 1] ) i++;
            right = B[p][i] + 1;
            if (A[left] < p && p > A[right]) now++;                      
            else if (A[left] > p && p < A[right]) now--;
        }
       ans = max(ans, now);//这个ans判断开始放到最里面了，导致70分，一个p的遍历过程可能有加有减
    }
    printf("%d\n", ans);
    return 0;
}

​ 这次写的比上一次更短，主要思想没变，但还是有小错误，要是考试真不一定能拿满。
