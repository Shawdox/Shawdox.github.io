CSP复盘202112-3-登机牌条码

​ 只考虑前40分，简单模拟即可，注意每个字符的编码前如果有状态转移，仅仅是多加了1-2个数字，vector+switch可以很好的避免使用过多的if-else：

// 202112-3 40分
#define _CRT_SECURE_NO_WARNINGS
#include "stdio.h"
#include "iostream"
#include "string"
#include "vector"
using namespace std;
int char_to_num(char c, int mode) {//mode = 0,1,2 大写，小写，数字
    if (mode == 0) return c - 'A';
    else if (mode == 1) return c - 'a';
    else if (mode == 2) return c - '0';
}
int char_to_mode(char c) {
    if ('A' <= c && c <= 'Z') return 0;
    else if ('a' <= c && c <= 'z') return 1;
    else if ('0' <= c && c <= '9') return 2;
}
void print_vec(vector<int> vi) {
    for (vector<int>::iterator it = vi.begin(); it != vi.end(); it++) printf("%d\n", *it);
}
void data_code(string str, int w) {
    vector<int> code_nums;//码数
    vector<int> code_bytes;//码字
    int mode = 0, cmode;
    for (string::iterator it = str.begin(); it != str.end(); it++) {
        cmode = char_to_mode(*it);//当前字符应置的模式
        switch (mode) {//用于模式转换
        case 0:
            if (cmode == 1) code_nums.push_back(27);
            else if (cmode == 2) code_nums.push_back(28);
            break;
        case 1:
            if (cmode == 0) code_nums.push_back(28), code_nums.push_back(28);//小写转大写，要换两次
            else if (cmode == 2) code_nums.push_back(28);
            break;
        case 2:
            if (cmode == 0) code_nums.push_back(28);
            else if (cmode == 1) code_nums.push_back(27);
            break;
        }
        code_nums.push_back(char_to_num(*it, cmode));//用于字符编码
        mode = cmode;
    }
    if (code_nums.size() % 2 != 0) code_nums.push_back(29);//补全
    for (int i = 0; i < code_nums.size() - 1; i += 2) code_bytes.push_back(code_nums[i] * 30 + code_nums[i + 1]);
    int k = (code_bytes.size() + 1) % w;
    int p = k==0 ? (code_bytes.size() + 1)/w : (code_bytes.size() + 1)/w + 1;
    if (k != 0)  for (int i = 0; i < w - k; ++i) code_bytes.push_back(900);//结尾补全使码字数+1整除w
   // print_vec(code_nums);
    printf("%d\n", p * w);
    print_vec(code_bytes);
}
int main(void) {
    int w, s;
    string input;
    scanf("%d %d", &w, &s);
    cin >> input;
    data_code(input, w);
    return 0;
}

​ 尤其注意p的取值会受到w的影响。
