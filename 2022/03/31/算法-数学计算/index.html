​ 仔细读题，这题并不是一个简单的模拟题，因为：

​ 故对于op=2的操作，没法直接除回去，这里可以想到不直接求模，而是求m的逆m^-1^ mod M：

​ 这样就可以了，但这就引出了另一个问题，M不一定是素数。故现有的定理得分情况讨论，这就变成了一道数论题，反而复杂化了。

​ 这里我们知道，对于某个op=1，m = m~k~的操作，我们需要求出$x = m{1}·m_2····m_k$;对于某个op=2，pos = i的操作，我们需要求出$x = m{1}·m_2····m_{i-1}·1·m_{i+1}·····m_k$ 。故完全可以建一颗N= Q，叶子结点都为1的线段树，op = 1时将对应的叶子结点改为m~k~，op = 2时将对应的叶子结点改为1即可，为了可读性这里没用位运算，写的比较长。

#include "stdio.h"
#include "string.h"
typedef long long ll;
#define MAX_M 100000
int t, Q, M;
struct node {
	ll l, r, pro = 1;
}tree[MAX_M << 2];
void build(ll node,ll L,ll R) {
	tree[node].l = L,tree[node].r = R;
	if (L == R){
		tree[node].pro = 1;
		return;
	}
	ll mid = (L + R) / 2;
	ll left_node = node * 2;
	ll right_node = node * 2 + 1;
	build(left_node, L, mid);
	build(right_node, mid + 1, R);
	tree[node].pro = 1;
}
void update(ll node, ll index, ll m) {
	if (tree[node].l == tree[node].r) {
		tree[node].pro = (tree[node].pro * m) % M;
		return;
	}
	ll mid = (tree[node].l + tree[node].r) / 2;
	if (index <= mid) update(node << 1, index, m);
	else update(node << 1 | 1, index, m);
	tree[node].pro = (tree[node << 1].pro * tree[node << 1 | 1].pro) % M;
}
void re_update(ll node, ll index) {
	if (tree[node].l == tree[node].r) {
		tree[node].pro = 1;
		return;
	}
	ll mid = (tree[node].l + tree[node].r) / 2;
	if (index <= mid) re_update(node << 1, index);
	else re_update(node << 1 | 1, index);
	tree[node].pro = (tree[node << 1].pro * tree[node << 1 | 1].pro) % M;
}
ll query(ll node, ll L, ll R) {
	if (L <= tree[node].l && tree[node].r <= R) return tree[node].pro;
	ll mid = (tree[node].l + tree[node].r) / 2;
	ll left_node = node * 2;
	ll right_node = node * 2 + 1;
	if (L > mid) return query(right_node, L, R);
	else if (R < mid + 1) return query(left_node, L, R);
	else {
		return (query(left_node,L,mid)*query(right_node,mid+1,R)) % M;
	}
}
int main(void) {
	scanf("%d", &t);
	for (int j = 0; j < t; ++j) {
		scanf("%d %d", &Q, &M);
		memset(tree, 0, sizeof(tree));
		build(1, 1, Q);
		for (int i = 1; i <= Q; ++i) {
			int op, m, pos;
			scanf("%d %d", &op,&m);
			if (op == 1) update(1, i, m);
			else if (op == 2) re_update(1, m);
               printf("%lld\n", tree[1].pro);
		}//end of i
	}//end of j
	return 0;
}//end of main
