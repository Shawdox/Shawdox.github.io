CSP复习-编码

【CPS复习】编码

#include "stdio.h"
#include "iostream"
#include "string"
using namespace std;
unsigned long long C(int n, int m) {
    unsigned long long ans = 1;
    for (int i = 1; i <= m; i++) ans = ans * (n - m + i) / i;
    return ans;
}
unsigned long long msum(int start, int end,int k) {
    unsigned long long ans = 0;
    if (start < 0 || end < 0) return 0;
    for (int i = start; i <= end; ++i) {
        ans += C(25 - i, k);
    }
    return ans;
}
int main(void) {
    string input;
    cin >> input;
    for (int i = 1; i < input.size(); ++i)
        if (input[i] <= input[i - 1]) {
            printf("0"); return 0;
        }
    int start = 0, end = 25;
    unsigned long long ans = 0;
    for (int j = 0; j < input.size(); ++j) {
        int k = input.size() - 1 - j;
        end = input[j] - 'a' - 1;
        ans += msum(start, end, k);
        start = end + 2;
    }
    for (int i = 1; i < input.size(); ++i) ans += C(26, i);
    printf("%llu\n", ans+1);
    return 0;
}
