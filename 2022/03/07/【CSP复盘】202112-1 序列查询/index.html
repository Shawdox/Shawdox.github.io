【CSP复盘】202112-1 序列查询

​ 考试的时候满脑子都是dp，当时想：

​ 然后用一个数组all_f[MAX_N]存储f(i)，然后dp：

//202112 -1
#include "stdio.h"
#include "string.h"
#define MAX_N 10000000
#define MAX_n 200
int n,N;
int A[MAX_n] = {0,};
int all_f[MAX_N];
int f(int i){
    if(all_f[i] >= 0) return all_f[i];
    else{
        if(A[f(i-1)] <= i && i < A[f(i-1)+1]){
            all_f[i] = f(i-1);
        }
        else if(f(i-1) == n){
            all_f[i] = f(i - 1);
        }
        else{
            all_f[i] = f(i - 1)+1;
        }
    return all_f[i];
    }
}
int main(){
    memset(all_f,-1,sizeof(all_f));
    all_f[0] = 0;
    scanf("%d %d",&n,&N);
    int ans = 0;
    for(int i = 1; i <= n;++i) scanf("%d",&A[i]);
    for(int j = 1; j < N;++j) ans += f(j);
    printf("%d\n",ans);
    return 0;
}

​ 但这样只能得40分，后来发现是MAX_n如果只设为200会导致溢出，应该大于等于201，更改后问题解决，100分通过，但做法还是一个一个计算f(i)，太耗费时间空间。

​ 但其实仔细观察f(i)的性质，就会发现完全没有必要这么麻烦：

​ 那完全可以用A[i]将连续数列0~N-1分段，且上一段的f(i)总是比下一段多1，相乘相加即可：

//202112 -1
#include "stdio.h"
int a[500] = {0};
int main(void){
    int n, N, ans = 0, t = 0;
    a[0] = 0;
    scanf("%d %d",&n,&N);
    for(int i = 1;i <= n;++i) scanf("%d",&a[i]);
    for(int j = 1;j <= n;++j) ans += (t++)*(a[j]-a[j-1]);
    ans += (N-a[n])*t;
    printf("%d\n",ans);
    return 0;
}
