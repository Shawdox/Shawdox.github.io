CSP复盘202112-2-序列查询新解

202112-2 序列查询新解

​ 注意到：

​ 要计算$error(A)$，如果将每一步的|g(i)-f(i)|都算出来则要计算10^9^次，但1s内最多算10^7^次。故如果不对最外层的大循环进行优化，无论对f(i)的计算如何优化结果都会超时，以下是70分代码：

// 202112-2 70分
#include "stdio.h"
using namespace std;
int n,N,a[300] = {0,};
int g(int x){
    int r = N/(n+1);
    return x/r;
}
int f(int x){//f(i)的计算是线性查询，也可以改写为在f(i-1)的基础上递推
    int ans = 0,i = 1;
    while(x >= a[i] && i <= n){
        ans++;
        i++;
    }
    return ans;
}
int abs(int x){
    return x<0?-1*x:x;
}
int main(void){
    scanf("%d %d",&n,&N);
    for(int i = 1;i <= n;++i)scanf("%d",&a[i]);
    //for(int i = 0;i < N;++i) printf("f(%d)=%d,g(%d)=%d\n",i,f(i),i,g(i));
    long long ans = 0;
    for(int i = 0;i < N;++i) ans += abs(f(i)-g(i));//这里大循环超时
    printf("%lld\n",ans);
    return 0;
}

​ 要拿满100分，就要对题目所给的”误差估计”方法入手（还是得数学优化）。
​ 故$f(i)$序列被a[i]分为好几个区间的同时，$g(i)$序列也被等距地分为好几个区间，在各自的区间内值不变。按照这个性质，我们可以固定每个$f(i)$的区间，在其中求每个$error(A)$：

  1. 若$f(i)\le g(i) or f(i)\ge g(i):$

    此区间内的$error(A)$可直接使用$|sum(f)-sum(g)|$来计算，由于f(i)已知且不变，故很方便计算。g(i)虽然已知但需要找到快速求$sum(g)$的方法，根据其类似等差数列的性质可以有：

    $g(i)$等于r个公差为1的等差数列叠加，可以方便得到$g(i)$的前缀和公式：

    故：

  2. 否则，找到区间内f(i)=g(i)的点，将区间一分为二，重复步骤一。

满分代码如下：

// 202112-2 100分
#include "stdio.h"
int n,N,a[100009] = {0,};
int r;
int abs(int x){
    return x < 0 ? -1 * x : x;
}
int g(int x){
    return x/r;
}
int f(int x){
    int ans = 0,i = 1;
    while(x >= a[i] && i <= n){
        ans++;i++;
    }
    return ans;
}
long long h(int x){
    if(x <0) return 0;
    int a = x/r;
    int b = x%r;
    return 0.5*a*(a-1)*r + (b+1)*a;
}
long long cal(int a,int b,int f){
    return abs((b - a + 1) * f - (h(b) - h(a-1)));
}
int main(void){
    scanf("%d %d",&n,&N);
    r = N / (n + 1);
    for(int i = 1;i <= n;++i)scanf("%d",&a[i]);
    long long ans = 0;
    for(int fi = 0;fi <= n;++fi){//对于每个fi
        int lf = a[fi];
        int rf = a[fi+1]-1;
        if(fi == n) rf = N-1;
        if(fi <= g(lf) || fi >= g(rf)){
            ans += cal(lf,rf,fi);
        }
        else{
            ans += cal(lf,r*fi,fi)+cal(r*fi +1,rf,fi)
        }
    }
    printf("%lld\n",ans);
    return 0;
}
