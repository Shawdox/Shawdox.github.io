CSP复习-组合数

【CPS复习】组合数

阶乘的计算方法：

  1. 套公式硬算：就算以long long 来存储数据，也只能存储到20！，这里m与n最高可以到2000，不可能套公式硬算。
  2. 递推：

​ 当m == n或者m== 0时组合数的值为1，此为递推边界，并且为了防止重复计算，可以得到以下递归代码：

long long res[67][67] = {0};
long long C(long long n,long long m){
     if(m == 0||n == 0) return 1;
     else if(res[m][n] == 0) res[m][n] == C(n-1,m)+C(n-1,m-1);
     else return res[m][n];
}

​ 算法复杂度为O(n^2^).

  1. 定义式变形：
long long C(long long n,long long m){
	long long ans = 1;
     for(long long i = 1;i <= m; ++i) ans *= (n-m+i)/i;
     return ans;
}

​ 算法复杂度为O(m)。

90分：

#include "stdio.h"
#include "string.h"
#include "algorithm"
using namespace std;
int t, k;
long long res[2001][2001];
long long C(int n, int m) {
    if (m == 0 || n == m) return 1;
    if (res[n][m] != -1) return res[n][m];
    res[n][m] = (C(n - 1, m) + C(n - 1, m - 1)) % k;
    return res[n][m];
}
bool is_ok(int a, int b) {
    if (C(a, b) == 0) return true;
    return false;
}
int ok_num(int n, int m) {
    int ans = 0;
    for (int i = 0; i <= n; ++i) {
        for (int j = 0; j <= min(i, m); ++j) {
            if (is_ok(i, j) == true) {
                ans++;
                //printf("i = %d, j = %d\n", i, j);
            }
        }
    }
    return ans;
}
int main(void) {
    memset(res, -1, sizeof(res));
    scanf("%d %d", &t, &k);
    int n, m;
    for (int i = 0; i < t; ++i) {
        scanf("%d %d", &n, &m);
        printf("%d\n", ok_num(n, m));
    }
    return 0;
}
