算法-校门外的区间

【算法复习】校门外的区间

​ 这题细节太多，做的时候快哭了。

一、解题思路：

​ 这题的思路并不是很复杂，只有两个方面：1. 集合运算到区间修改的转换；2. 如何用离散的整数表示连续的区间。

​ 对于前者，简单画个示意图就可以得出：

① U T： T全部置1；
② I T： T外区间全部置0；
③ D T： T全部置0；
④ C T:  T外区间置0，T内区间翻转；
⑤ S T： T内区间翻转。

​ 这样主需要完成区间覆盖与区间翻转即可，注意每次change时先覆盖，再翻转。

二、注意事项：

  1. 输出结果之前需要将所有的tag下放；

  2. 真实下标与线段树中下标的转换：

    2.1 一般线段树以1开头，这里区间可以为0；

    2.2 线段树下标是通过开闭区间+真实下标计算的；

  3. 可能出现一个小区间的线段树下标左边>右边的情况（（3,3），（3，3]），要直接屏蔽，不然RE；

三、AC代码：

#define MAX_N 65536
#include "bits/stdc++.h"
using namespace std;
typedef long long ll;
struct Node {
	ll l, r, cov, rev;
}tree[MAX_N << 3];
int ans[MAX_N<<1+5];
void Build(int node, int L, int R) {
	tree[node].l = L, tree[node].r = R,tree[node].cov = -1;
	if (L == R) return;
	Build(node << 1, L, (L + R) / 2);
	Build(node << 1 | 1, (L + R) / 2 + 1, R);
}
void change(int node) {
	if (tree[node].l != tree[node].r) {
		if (tree[node].cov != -1) {
			tree[node << 1].cov = tree[node << 1 | 1].cov = tree[node].cov;
			tree[node << 1].rev = tree[node << 1 | 1].rev = 0;
			tree[node].cov = -1;
		}
		if (tree[node].rev) {
			tree[node << 1].rev ^= 1;
			tree[node << 1 | 1].rev ^= 1;
			tree[node].rev = 0;
		}
	}
}
void Update(int node, int L, int R, int mode,int vul) {//mode = 0,cov,1,rev
	change(node);
	if (L <= tree[node].l && tree[node].r <= R) {
		if (mode == 0) {
			tree[node].cov = vul;
			tree[node].rev = 0;
		}
		else tree[node].rev ^= 1;
		return;
	}
	int mid = (tree[node].l + tree[node].r) / 2;
	if (R < mid + 1) Update(node << 1, L, R, mode, vul);
	else if (L > mid) Update(node << 1 | 1, L, R, mode, vul);
	else {
		Update(node << 1, L, mid, mode, vul);
		Update(node << 1|1, mid+1, R, mode, vul);
	}
}
int query(int node, int index) {
	change(node);
	if (tree[node].l == tree[node].r) {
		if (tree[node].cov == -1) return tree[node].rev;
		else return tree[node].cov^ tree[node].rev;
	}
	int mid = (tree[node].l + tree[node].r) / 2;
	if (index < mid + 1) return query(node << 1, index);
	else return query(node << 1 | 1, index);
}

void mprint() {
	int start, end;
	string output = "";
	for (int i = 1; i <= MAX_N<<1 ; ++i) ans[i] = query(1, i);//将所有tag下放
	for (int n = 1; n <= MAX_N<<1 ; ++n) {
		int t = ans[n];
		while (n <= MAX_N<<1 && !t) t = ans[++n];
		start = n;
		if (start >= MAX_N << 1 && t == 0) break;
		while(n <= MAX_N << 1 && t) t =ans[(n++)+1];
		end = --n;
		
		if (start % 2 ==0 ) {
			output += "[";
			output += to_string(start / 2 -1);//数可能不止一位
		}
		else {
			output += "(";
			output += to_string((start - 1) / 2 -1);
		}
		output += ',';
		if (end % 2 == 0) {
			output += to_string(end / 2 -1);
			output += ']';
		}
		else {
			output += to_string((end + 1) / 2 -1);
			output += ')';
		}
		output += ' ';
	}
	if (output == "") printf("empty set\n");
	else cout << output << endl;
}
void str_to_int(string s,int *start,int *end) {
	int ans1 = 0,ans2 = 0;
	int i;
	for (i = 1; s[i] != ','; ++i) 
		ans1 = ans1 * 10 + s[i] - '0';
	++i;
	for (; i < s.size()-1; ++i)
		ans2 = ans2 * 10 + s[i] - '0';
	*start = ans1, *end = ans2;
}

int main(void) {
	Build(1, 1, MAX_N << 1);
	char c;
	string input;
	int start, end;
	while ((c = getchar()) != EOF) {
		cin >> input;
		getchar();
		str_to_int(input, &start, &end);
		start = 2 * (start + 1);//题目从零开始，建树从1开始
		end = 2 * (end + 1);
		if (input[0] == '(') start++;
		if (input[input.size()-1] == ')') end--;
		if (start > end) continue;//这里尤其注意
		switch (c){
		case 'U':
			Update(1, start, end, 0, 1);
			break;
		case 'I':
			if (start > 1) Update(1, 1, start - 1, 0, 0);
			if (end < MAX_N << 1) Update(1, end + 1, MAX_N << 1, 0, 0);
			break;
		case 'D':
			Update(1, start, end, 0, 0);
			break;
		case 'C':
			if (start > 1) Update(1, 1, start - 1, 0, 0);
			if (end < MAX_N << 1) Update(1, end + 1, MAX_N << 1, 0, 0);
			Update(1, start, end, 1, 0);
			break;
		case 'S':
			Update(1, start, end, 1, 0);
			break;
		}
	}
	mprint();
	return 0;
}
