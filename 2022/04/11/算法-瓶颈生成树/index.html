算法-瓶颈生成树

【算法复习】瓶颈生成树

1. 什么是瓶颈生成树？

2. 此题思路：

​ 了解了瓶颈生成树的概念后，就可以发现，只需要求解最小生成树，然后去掉前S-1大的边（因为卫星电话可以给谁不是固定的，所以给最大的边）即可。

#include "bits/stdc++.h"
using namespace std;
int parents[501];
int find_parents(int x) {
	int a = x;
	while (x != parents[x]) x = parents[x];
	while (a != parents[a]) {
		int z = a;
		a = parents[a];
		parents[z] = x;
	}
	return x;
}
struct Edge {
	int x, y;
	double dis;
};
vector<Edge> E;
vector<pair<int, int>> Node;
bool cmp(struct Edge a, struct Edge b) { return a.dis < b.dis; }
double kruskal(int N,int M,int s) {
	for (int i = 0; i <= N; ++i) parents[i] = i;
	int cnt = 0;
	vector<double> ans;
	vector<Edge>::iterator it = E.begin();
	sort(it, it + M, cmp);
	for (int i = 0; i < M; ++i) {
		int u = E[i].x, v = E[i].y;
		int pu = find_parents(u), pv = find_parents(v);
		if (pu != pv) {
			parents[pu] = pv;
			ans.push_back(E[i].dis);
			cnt++;
			if (cnt == N - 1) break;
		}
	}
	if (cnt != N - 1) return -1;
	else return ans[N - s - 2];
}
int main(void) {
	int S, P, x, y;
	scanf("%d %d", &S, &P);
	for (int i = 1; i <= P; ++i) {
		scanf("%d %d", &x, &y);
		for (int j = 0; j < Node.size(); ++j) {
			struct Edge p = {i,j+1,sqrt((x-Node[j].first)* (x - Node[j].first) + (y - Node[j].second)* (y - Node[j].second))};
			E.push_back(p);
		}
		Node.push_back(make_pair ( x, y ));
	}
	printf("%.2lf", kruskal(P, E.size(), S-1));
	return 0;
}
